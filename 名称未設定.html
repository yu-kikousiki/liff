  <h1>使っている楽器を選んでください</h1>

  <button onclick="selectInstrument('サックス')">サックス</button>

  <button onclick="selectInstrument('フルート')">フルート</button>

  <button onclick="selectInstrument('クラリネット')">クラリネット</button>

  <button onclick="selectInstrument(‘ホルン)”>ホルン</button>

  <button onclick="selectInstrument(‘トランペット’)”>トランペット</button>

  <button onclick="selectInstrument(‘トロンボーン)”>トロンボーン</button>

  <button onclick="selectInstrument(‘ユーフォニアム)”>ユーフォニアム</button>

  <button onclick="selectInstrument(‘チューバ’)”>チューバ</button>

  <button onclick="selectInstrument(‘コントラバス’)”>コントラバス</button>

  <button onclick="selectInstrument(‘パーカッション’)”>パーカッション</button>


  <script>

    // LIFF初期化

    liff.init({ liffId: "あなたのLIFF IDをここに" })

      .then(() => {

        if (!liff.isLoggedIn()) {

          liff.login();

        }

      });


    // 楽器を選んだ時の処理

    function selectInstrument(instrument) {

      // 選択結果をBotに送信（トーク画面にテキストを送る）

      liff.sendMessages([

        {

          type: "text",

          text: "楽器選択：" + instrument

        }

      ]).then(() => {

        alert(instrument + "を選びました！");

        liff.closeWindow(); // ウィンドウを閉じる

      }).catch((err) => {

        alert("送信エラー: " + err);

      });

    }

  </script>
